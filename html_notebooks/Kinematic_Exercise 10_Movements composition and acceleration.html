# Bour formula:
# V_02_B0 = V_02_B2 + R_02 ^ OM_B2
# G_02 = d(V_02_B0)/dt
# G_02 = d(V_02_B1)/dt_B1 + R_01 ^ V_02_B0

R_01 = B1.ang_vel_in(B0).to_matrix(B0)
R_12 = B2.ang_vel_in(B1).to_matrix(B0)
R_02 = R_01 + R_12  # Using Sympy method: B2.ang_vel_in(B0).to_matrix(B0)
